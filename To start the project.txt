GRANT ALL PRIVILEGES ON ALL TABLES IN SCHEMA public TO wehirenow_user;
GRANT ALL PRIVILEGES ON ALL SEQUENCES IN SCHEMA public TO wehirenow_user;
GRANT USAGE ON SCHEMA public TO wehirenow_user;

-- 4. Pour les tables spécifiques :
GRANT SELECT, INSERT, UPDATE, DELETE ON candidate_profiles TO wehirenow_user;
GRANT SELECT, INSERT, UPDATE, DELETE ON recruiter_profiles TO wehirenow_user;
GRANT SELECT, INSERT, UPDATE, DELETE ON users TO wehirenow_user;
GRANT SELECT, INSERT, UPDATE, DELETE ON jobs TO wehirenow_user;
GRANT SELECT, INSERT, UPDATE, DELETE ON job_applications TO wehirenow_user;

-- 5. Donner aussi les droits sur les séquences (pour les auto-incréments) :
GRANT USAGE, SELECT ON ALL SEQUENCES IN SCHEMA public TO wehirenow_user;

-- Spécifiquement pour chaque séquence :
GRANT USAGE, SELECT ON SEQUENCE users_id_seq TO wehirenow_user;
GRANT USAGE, SELECT ON SEQUENCE jobs_id_seq TO wehirenow_user;
GRANT USAGE, SELECT ON SEQUENCE job_applications_id_seq TO wehirenow_user;
GRANT USAGE, SELECT ON SEQUENCE candidates_id_seq TO wehirenow_user;
GRANT USAGE, SELECT ON SEQUENCE recruiters_id_seq TO wehirenow_user;
